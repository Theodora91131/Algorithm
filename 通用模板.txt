void dfs(int index, 狀態資料...) {
    if (達成目標) {
        保存結果;
        return;
    }
    for (int i = index; i < 候選項目.size(); ++i) {
        if (條件不符) continue;
        做選擇;          
        dfs(i 或 i+1, 更新狀態);
        撤銷選擇;
    }
}

int gcd(int p, int q) {
    if(q == 0){
        return p;
    } 
    return gcd(q,p % q);
}

int lcm(int p, int q) {
    return p / gcd(p,q)*q;
}

#include<bits/stdc++.h>

const int MaxN = 1000005;
int fact[MaxN];
using namespace std;
int main(){
	int n, cnt ,p;
	for(long long i =2; i<=MaxN;i++){
		if(!fact[i]){
			fact[i]=i;
			for(long long j=i; j<=MaxN; j+=i){
				if(!fact[j]){
					fact[j]=i;
				}
			}
		}
	}
	while(cin>>n && n){
		int temp = n;
		int cnt = 0;
		while(temp != 1){
			int p = fact[temp];
			cnt++;
			while(temp %p ==0){
				temp /=p;
			}
		}
		cout << n <<" : "<<cnt<<endl;
	} 


	return 0;
}
